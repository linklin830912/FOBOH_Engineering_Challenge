import { useMemo, useState } from "react";

export type Product = {

  sku: string;

  segment: string;
  brand: string;

type Props = {
  products: Product[];
  onFiltered?: (filtered: Product[]) => void;
};

export default function ProductSearchFilter({
  products,
  onFiltered,
}: Props) {
  const [nameSearch, setNameSearch] = useState("");
  const [skuSearch, setSkuSearch] = useState("");
  const [category, setCategory] = useState("");
  const [segment, setSegment] = useState("");
  const [brand, setBrand] = useState("");

  const filteredProducts = useMemo(() => {
    const result = products.filter((p) => {
      const matchName =
        p.name.toLowerCase().includes(nameSearch.toLowerCase());

      const matchSku =
        p.sku.toLowerCase().includes(skuSearch.toLowerCase());

      const matchCategory = category ? p.category === category : true;
      const matchSegment = segment ? p.segment === segment : true;
      const matchBrand = brand ? p.brand === brand : true;

      return matchName && matchSku && matchCategory && matchSegment && matchBrand;
    });

    onFiltered?.(result);
    return result;
  }, [products, nameSearch, skuSearch, category, segment, brand, onFiltered]);

  return (
    <div className="flex flex-col gap-4 w-full">
      {/* Search Row */}
      <div className="flex gap-4 flex-wrap">

        {/* Product Name Search */}
        <input
          className="w-[240px] border rounded-md px-3 py-2 text-sm"
          placeholder="Search Product Name"
          value={nameSearch}
          onChange={(e) => setNameSearch(e.target.value)}
        />

        {/* SKU Search */}
        <input
          className="w-[240px] border rounded-md px-3 py-2 text-sm"
          placeholder="Search SKU"
          value={skuSearch}
          onChange={(e) => setSkuSearch(e.target.value)}
        />

        {/* Category Filter */}
        <input
          className="w-[160px] border rounded-md px-3 py-2 text-sm"
          placeholder="Category"
          value={category}
          onChange={(e) => setCategory(e.target.value)}
        />

        {/* Segment Filter */}
        <input
          className="w-[160px] border rounded-md px-3 py-2 text-sm"
          placeholder="Segment"
          value={segment}
          onChange={(e) => setSegment(e.target.value)}
        />

        {/* Brand Filter */}
        <input
          className="w-[160px] border rounded-md px-3 py-2 text-sm"
          placeholder="Brand"
          value={brand}
          onChange={(e) => setBrand(e.target.value)}
        />
      </div>

      {/* Results (optional preview) */}
      <div className="text-sm text-gray-600">
        Showing {filteredProducts.length} products
      </div>
    </div>
  );
}

Here’s the clean step-by-step to set up a .env file in a React app (works for CRA or Vite—I'll show both since people often mix them up).

✅ 1. Create .env file (in project root)

At the same level as package.json:

.env
✅ 2. Add your variable
👉 If you're using CRA (Create React App)

You MUST prefix with REACT_APP_

REACT_APP_API_URL=http://localhost:3000
👉 If you're using Vite (recommended if newer setup)

You MUST prefix with VITE_

VITE_API_URL=http://localhost:3000
✅ 3. Restart dev server (IMPORTANT)

After adding .env:

npm run dev
# or
npm start

If you don’t restart, env won’t load.

✅ 4. Use it in code
👉 CRA
const API_URL = process.env.REACT_APP_API_URL;
👉 Vite
const API_URL = import.meta.env.VITE_API_URL;
⚠️ Common mistakes (your future self will thank you)
❌ forgetting REACT_APP_ or VITE_
❌ not restarting server
❌ putting quotes incorrectly ("http://..." not needed)
❌ trying to use .env inside backend logic (React only sees client-side vars)

// src/config/env.ts
export const env = {

} from "./db";

/**
 * GET /pricing-profile/match?customerId=xxx&productId=yyy
 */
app.get("/pricing-profile/match", (req, res) => {
  const { customerId, productId } = req.query;

  // =================================================
  // 1. VALIDATION
  // =================================================
  if (!customerId || !productId) {
    return res.status(400).json({
      status: "error",
      message: "customerId and productId are required",
    });
  }

  // =================================================
  // 2. FIND CUSTOMER
  // =================================================
  const customer = MOCK_CUSTOMERS_STORE.find(
    (c) => c.id === customerId
  );

  if (!customer) {
    return res.status(404).json({
      status: "error",
      message: "Customer not found",
    });
  }

  const customerGroupIds = customer.groupIds ?? [];

  // =================================================
  // 3. FIND GROUPS (optional but useful for debugging)
  // =================================================
  const groups = MOCK_CUSTOMER_GROUP_STORE.filter((g) =>
    customerGroupIds.includes(g.id)
  );

  // =================================================
  // 4. FIND MATCHING PRICING PROFILES
  // =================================================
  const matchedProfiles = MOCK_PRICING_PROFILES_STORE.filter(
    (profile) => {
      const hasCustomerGroupMatch = profile.customerGroupIds?.some((id: string) =>
        customerGroupIds.includes(id)
      );

      const hasProductMatch = profile.productIds?.includes(
        productId as string
      );

      return hasCustomerGroupMatch && hasProductMatch;
    }
  );

  // =================================================
  // 5. SORT BY PRIORITY (IMPORTANT FOR PRICING ENGINE)
  // =================================================
  matchedProfiles.sort((a, b) => (b.priority ?? 0) - (a.priority ?? 0));

  // =================================================
  // 6. RESPONSE
  // =================================================
  return res.json({
    status: "ok",
    data: {
      customerId,
      productId,
      groups,
      matchedProfiles,
      bestMatch: matchedProfiles[0] ?? null,
    },
  });
});

      return hasGroupMatch && hasProductMatch;

  // =================================================
  // 4. SORT (priority DESC → price ASC)

  const sorted = matchedProfiles.sort((a, b) => {
